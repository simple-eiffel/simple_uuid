			-- And does the string s correspond to code_string?
		require

			-- To create an instance of type DATE_TIME
			-- And does the string s correspond to code_string?

			s_exists: s /= Void
			code_exists: code_string /= Void
			base_valid: base > 0 and (base \\ 100 = 0)
